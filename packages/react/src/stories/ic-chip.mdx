    },
    {
      firstName: "Naomi",

      age: 32,
    },
    {
      firstName: "Luke",
      lastName: "Ashford",
      jobTitle: "Junior Developer",
      age: 18,
    },
  ];

  const colDefs = [
    { field: "firstName" },
    { field: "lastName" },
    { field: "jobTitle", cellRenderer: JobTitleRenderer },
    { field: "age" },
  ];

  return (
    <>
      <div
        className="ag-theme-quartz"
      >
        <AgGridReact
          rowData={rowData}
          columnDefs={colDefs}
          domLayout="autoHeight"
        />
      </div>
    </>
  )
}
